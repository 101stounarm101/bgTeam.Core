namespace $namespace$
{
    /// <summary>
    /// Entrypoint class for application.
    /// </summary>
    internal static class Program
    {
        /// <summary>
        /// Entrypoint method for application.
        /// </summary>
        /// <param name="args"></param>
        internal static void Main(string[] args)
        {
            new ApplicationBuilder(args)
                .BuildApplication()
                .StartHealthChecks()
                .RunServices()
                .WaitForExit();
        }
    }
}
