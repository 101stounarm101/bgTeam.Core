namespace $namespace$
{
    using bgTeam.Core.Helpers;
    using Microsoft.Extensions.DependencyInjection;

    /// <summary>
    /// Startup for console app.
    /// </summary>
    public class ApplicationBuilder
    {
        /// <summary>
        /// Initializes a new instance of the <see cref="ApplicationBuilder"/> class.
        /// Create service collection using env variables.
        /// </summary>
        /// <param name="args"></param>
        public ApplicationBuilder(string[] args)
        {
            var cmdParams = CommandLineHelper.ParseArgs(args);
            ServiceCollection = AppIocConfigure.Configure(cmdParams);
        }

        /// <summary>
        /// Service collection for container.
        /// </summary>
        public IServiceCollection ServiceCollection { get; }

        /// <summary>
        /// Returns instance of built application.
        /// </summary>
        /// <returns></returns>
        public Application BuildApplication()
        {
            return new Application(ServiceCollection.BuildServiceProvider());
        }
    }
}
